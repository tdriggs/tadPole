623  typedef map<size_t, size_t>::type IndexRemap;
628  template <typename T>
629  void buildIndexRemap(T* pBuffer, size_t numIndexes, IndexRemap& remap)
630  {
631  remap.clear();
632  for (size_t i = 0; i < numIndexes; ++i)
633  {
634  // use insert since duplicates are silently discarded
635  remap.insert(IndexRemap::value_type(*pBuffer++, remap.size()));
636  // this will have mapped oldindex -> new index IF oldindex
637  // wasn't already there
638  }
639  }
641  template <typename T>
642  void remapIndexes(T* src, T* dst, const IndexRemap& remap,
643  size_t numIndexes)
644  {
645  for (size_t i = 0; i < numIndexes; ++i)
646  {
647  // look up original and map to target
648  IndexRemap::const_iterator ix = remap.find(*src++);
649  assert(ix != remap.end());
650  *dst++ = static_cast<T>(ix->second);
651  }
652  }

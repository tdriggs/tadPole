37 // Because there are more than 32 possible Capabilities, more than 1 int is needed to store them all.
38 // In fact, an array of integers is used to store capabilities. However all the capabilities are defined in the single
39 // enum. The only way to know which capabilities should be stored where in the array is to use some of the 32 bits
40 // to record the category of the capability. These top few bits are used as an index into mCapabilities array
41 // The lower bits are used to identify each capability individually by setting 1 bit for each
42 
43 // Identifies how many bits are reserved for categories
44 // NOTE: Although 4 bits (currently) are enough
45 #define CAPS_CATEGORY_SIZE 4
46 #define OGRE_CAPS_BITSHIFT (32 - CAPS_CATEGORY_SIZE)
47 #define CAPS_CATEGORY_MASK (((1 << CAPS_CATEGORY_SIZE) - 1) << OGRE_CAPS_BITSHIFT)
48 #define OGRE_CAPS_VALUE(cat, val) ((cat << OGRE_CAPS_BITSHIFT) | (1 << val))

246  struct _OgreExport DriverVersion
247  {
248  int major;
249  int minor;
250  int release;
251  int build;
252 
253  DriverVersion()
254  {
255  major = minor = release = build = 0;
256  }

258  String toString() const
259  {
260  StringStream str;
261  str << major << "." << minor << "." << release << "." << build;
262  return str.str();
263  }
264 
265  void fromString(const String& versionString)
266  {
267  StringVector tokens = StringUtil::split(versionString, ".");
268  if(!tokens.empty())
269  {
270  major = StringConverter::parseInt(tokens[0]);
271  if (tokens.size() > 1)
272  minor = StringConverter::parseInt(tokens[1]);
273  if (tokens.size() > 2)
274  release = StringConverter::parseInt(tokens[2]);
275  if (tokens.size() > 3)
276  build = StringConverter::parseInt(tokens[3]);
277  }
278 
279  }
280  };

553  void setCapability(const Capabilities c)
554  {
555  int index = (CAPS_CATEGORY_MASK & c) >> OGRE_CAPS_BITSHIFT;
556  // zero out the index from the stored capability
557  mCapabilities[index] |= (c & ~CAPS_CATEGORY_MASK);
558  }
559 
562  void unsetCapability(const Capabilities c)
563  {
564  int index = (CAPS_CATEGORY_MASK & c) >> OGRE_CAPS_BITSHIFT;
565  // zero out the index from the stored capability
566  mCapabilities[index] &= (~c | CAPS_CATEGORY_MASK);
567  }
568 
571  bool hasCapability(const Capabilities c) const
572  {
573  int index = (CAPS_CATEGORY_MASK & c) >> OGRE_CAPS_BITSHIFT;
574  // test against
575  if(mCapabilities[index] & (c & ~CAPS_CATEGORY_MASK))
576  {
577  return true;
578  }
579  else
580  {
581  return false;
582  }

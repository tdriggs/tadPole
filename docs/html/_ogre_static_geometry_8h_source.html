560  typedef map<size_t, size_t>::type IndexRemap;
565  template <typename T>
566  void buildIndexRemap(T* pBuffer, size_t numIndexes, IndexRemap& remap)
567  {
568  remap.clear();
569  for (size_t i = 0; i < numIndexes; ++i)
570  {
571  // use insert since duplicates are silently discarded
572  remap.insert(IndexRemap::value_type(*pBuffer++, remap.size()));
573  // this will have mapped oldindex -> new index IF oldindex
574  // wasn't already there
575  }
576  }
578  template <typename T>
579  void remapIndexes(T* src, T* dst, const IndexRemap& remap,
580  size_t numIndexes)
581  {
582  for (size_t i = 0; i < numIndexes; ++i)
583  {
584  // look up original and map to target
585  IndexRemap::const_iterator ix = remap.find(*src++);
586  assert(ix != remap.end());
587  *dst++ = static_cast<T>(ix->second);
588  }
589  }

599  inline Vector3 randomDeviant(
600  const Radian& angle,
601  const Vector3& up = Vector3::ZERO ) const
602  {
603  Vector3 newUp;
604 
605  if (up == Vector3::ZERO)
606  {
607  // Generate an up vector
608  newUp = this->perpendicular();
609  }
610  else
611  {
612  newUp = up;
613  }
614 
615  // Rotate up vector by random amount around this
616  Quaternion q;
617  q.FromAngleAxis( Radian(Math::UnitRandom() * Math::TWO_PI), *this );
618  newUp = q * newUp;
619 
620  // Finally rotate this by given angle around randomised up
621  q.FromAngleAxis( angle, newUp );
622  return q * (*this);
623  }

629  inline Radian angleBetween(const Vector3& dest) const
630  {
631  Real lenProduct = length() * dest.length();
632 
633  // Divide by zero check

651  Quaternion getRotationTo(const Vector3& dest,
652  const Vector3& fallbackAxis = Vector3::ZERO) const
653  {
654  // From Sam Hocevar's article "Quaternion from two vectors:
655  // the final version"
656  Real a = Math::Sqrt(this->squaredLength() * dest.squaredLength());
657  Real b = a + this->dotProduct(dest);
658  Vector3 axis;
659 
660  if (b < (Real)1e-06 * a)
661  {
662  b = (Real)0.0;
663  axis = fallbackAxis != Vector3::ZERO ? fallbackAxis
664  : Math::Abs(x) > Math::Abs(z) ? Vector3(-y, x, (Real)0.0)
665  : Vector3((Real)0.0, -z, y);
666  }
667  else
668  {
669  axis = this->crossProduct(dest);
670  }
671 
672  Quaternion q(b, axis.x, axis.y, axis.z);
673  q.normalise();
674  return q;
675  }

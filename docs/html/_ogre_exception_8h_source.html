269  static OGRE_NORETURN void _throwException(
270  Exception::ExceptionCodes code, int number,
271  const String& desc,
272  const String& src, const char* file, long line)
273  {
274  switch (code)
275  {
276  case Exception::ERR_CANNOT_WRITE_TO_FILE: throw IOException(number, desc, src, file, line);
277  case Exception::ERR_INVALID_STATE: throw InvalidStateException(number, desc, src, file, line);
278  case Exception::ERR_INVALIDPARAMS: throw InvalidParametersException(number, desc, src, file, line);
279  case Exception::ERR_RENDERINGAPI_ERROR: throw RenderingAPIException(number, desc, src, file, line);
280  case Exception::ERR_DUPLICATE_ITEM: throw ItemIdentityException(number, desc, src, file, line);
281  case Exception::ERR_ITEM_NOT_FOUND: throw ItemIdentityException(number, desc, src, file, line);
282  case Exception::ERR_FILE_NOT_FOUND: throw FileNotFoundException(number, desc, src, file, line);
283  case Exception::ERR_INTERNAL_ERROR: throw InternalErrorException(number, desc, src, file, line);
284  case Exception::ERR_RT_ASSERTION_FAILED: throw RuntimeAssertionException(number, desc, src, file, line);
285  case Exception::ERR_NOT_IMPLEMENTED: throw UnimplementedException(number, desc, src, file, line);
286  case Exception::ERR_INVALID_CALL: throw InvalidCallException(number, desc, src, file, line);
287  default: throw Exception(number, desc, src, "Exception", file, line);
288  }
289  }
290  public:
291  static OGRE_NORETURN void throwException(
292  Exception::ExceptionCodes code,
293  const String& desc,
294  const String& src, const char* file, long line)
295  {
296  _throwException(code, code, desc, src, file, line);
297  }

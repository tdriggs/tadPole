160  // HACK: Multi-index emulation container to store constant buffer information by index and name at same time
161  // using tips from http://www.boost.org/doc/libs/1_35_0/libs/multi_index/doc/performance.html
162  // and http://cnx.org/content/m35767/1.2/
163 #define INVALID_IDX (unsigned int)-1
164  struct BufferInfo
165  {
166  static _StringHash mHash;
167  unsigned int mIdx;
168  String mName;
169  mutable HardwareUniformBufferSharedPtr mUniformBuffer;
170  mutable ShaderVars mShaderVars;
171 
172  // Default constructor
173  BufferInfo() : mIdx(0), mName("") { mUniformBuffer.reset(); }
174  BufferInfo(unsigned int index, const String& name)
175  : mIdx(index), mName(name)
176  {
177  mUniformBuffer.reset();
178  }
179 
180  // Copy constructor
181  BufferInfo(const BufferInfo& info)
182  : mIdx(info.mIdx)
183  , mName(info.mName)
184  , mUniformBuffer(info.mUniformBuffer)
185  , mShaderVars(info.mShaderVars)
186  {
187 
188  }
189 
190  // Copy operator
191  BufferInfo& operator=(const BufferInfo& info)
192  {
193  this->mIdx = info.mIdx;
194  this->mName = info.mName;
195  mUniformBuffer = info.mUniformBuffer;
196  mShaderVars = info.mShaderVars;
197  return *this;
198  }
199 
200  // Constructors and operators used for search
201  BufferInfo(unsigned int index) : mIdx(index), mName("") { }
202  BufferInfo(const String& name) : mIdx(INVALID_IDX), mName(name) { }
203  BufferInfo& operator=(unsigned int index) { this->mIdx = index; return *this; }
204  BufferInfo& operator=(const String& name) { this->mName = name; return *this; }
205 
206  bool operator==(const BufferInfo& other) const
207  {
208  return mName == other.mName && mIdx == other.mIdx;
209  }
210  bool operator<(const BufferInfo& other) const
211  {
212  if (mIdx == INVALID_IDX || other.mIdx == INVALID_IDX)
213  {
214  return mName < other.mName;
215  }
216  else if (mName == "" || other.mName == "")
217  {
218  return mIdx < other.mIdx;
219  }
220  else
221  {
222  if (mName == other.mName)
223  {
224  return mIdx < other.mIdx;
225  }
226  else
227  {
228  return mName < other.mName;
229  }
230  }
231  }
232  };

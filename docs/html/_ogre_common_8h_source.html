570  template< typename T > struct TRect
571  {
572  T left, top, right, bottom;
573  TRect() : left(0), top(0), right(0), bottom(0) {}
574  TRect( T const & l, T const & t, T const & r, T const & b )
575  : left( l ), top( t ), right( r ), bottom( b )
576  {
577  }
578  TRect( TRect const & o )
579  : left( o.left ), top( o.top ), right( o.right ), bottom( o.bottom )
580  {
581  }
582  TRect & operator=( TRect const & o )
583  {
584  left = o.left;
585  top = o.top;
586  right = o.right;
587  bottom = o.bottom;
588  return *this;
589  }
590  T width() const
591  {
592  return right - left;
593  }
594  T height() const
595  {
596  return bottom - top;
597  }
598  bool isNull() const
599  {
600  return width() == 0 || height() == 0;
601  }
602  void setNull()
603  {
604  left = right = top = bottom = 0;
605  }
606  TRect & merge(const TRect& rhs)
607  {
608  if (isNull())
609  {
610  *this = rhs;
611  }
612  else if (!rhs.isNull())
613  {
614  left = std::min(left, rhs.left);
615  right = std::max(right, rhs.right);
616  top = std::min(top, rhs.top);
617  bottom = std::max(bottom, rhs.bottom);
618  }
619 
620  return *this;
621 
622  }
623  TRect intersect(const TRect& rhs) const
624  {
625  TRect ret;
626  if (isNull() || rhs.isNull())
627  {
628  // empty
629  return ret;
630  }
631  else
632  {
633  ret.left = std::max(left, rhs.left);
634  ret.right = std::min(right, rhs.right);
635  ret.top = std::max(top, rhs.top);
636  ret.bottom = std::min(bottom, rhs.bottom);
637  }
638 
639  if (ret.left > ret.right || ret.top > ret.bottom)
640  {
641  // no intersection, return empty
642  ret.left = ret.top = ret.right = ret.bottom = 0;
643  }
644 
645  return ret;
646 
647  }

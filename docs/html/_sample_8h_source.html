58  | Utility comparison structure for sorting samples using SampleSet.

60  struct Comparer
61  {
62  bool operator() (Sample* a, Sample* b)
63  {
64  Ogre::NameValuePairList::iterator aTitle = a->getInfo().find("Title");
65  Ogre::NameValuePairList::iterator bTitle = b->getInfo().find("Title");
66 
67  if (aTitle != a->getInfo().end() && bTitle != b->getInfo().end())
68  return aTitle->second.compare(bTitle->second) < 0;
69  else return false;
70  }

163  {

177 #endif

179  mRoot->destroySceneManager(mSceneMgr);
180  }

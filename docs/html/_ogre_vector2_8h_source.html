504  inline Ogre::Radian angleBetween(const Ogre::Vector2& other) const
505  {
506  Ogre::Real lenProduct = length() * other.length();
507  // Divide by zero check
508  if(lenProduct < 1e-6f)
509  lenProduct = 1e-6f;
510 
511  Ogre::Real f = dotProduct(other) / lenProduct;
512 
513  f = Ogre::Math::Clamp(f, (Ogre::Real)-1.0, (Ogre::Real)1.0);
514  return Ogre::Math::ACos(f);
515  }

83  Ogre::String::size_type pos = filename.rfind('.');
84  if (pos != Ogre::String::npos)
85  filename = filename.substr(0, pos) + OGRE_BUILD_SUFFIX + filename.substr(pos);

88  // look for the requested file in several locations:
89 
90  // 1. in the writable path (so user can provide custom files)
91  Ogre::String path = getWritablePath(filename);
92  if (fileExists(path))
93  return path;
94 
95  // 2. in the config file search paths
96  for (size_t i = 0; i < mConfigPaths.size(); ++i)
97  {
98  path = mConfigPaths[i] + filename;
99  if (fileExists(path))
100  return path;
101  }
102 
103  // 3. fallback to current working dir
104  return filename;
105  }

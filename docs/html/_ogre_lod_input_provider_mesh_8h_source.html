63  template<typename IndexType>
64  void addIndexDataImpl(LodData* data, IndexType* iPos, const IndexType* iEnd,
65  VertexLookupList& lookup, size_t submeshID, RenderOperation::OperationType renderOp)
66  {
67  if(iEnd - iPos < 3
68  || (renderOp != RenderOperation::OT_TRIANGLE_LIST
69  && renderOp != RenderOperation::OT_TRIANGLE_STRIP
70  && renderOp != RenderOperation::OT_TRIANGLE_FAN))
71  return;
72 
73  IndexType i0 = iPos[0], i1 = iPos[1], i2 = iPos[2];
74  unsigned inc = (renderOp==RenderOperation::OT_TRIANGLE_LIST) ? 3 : 1;
75  unsigned triangleIdx = 0;
76 
77  // Loop through all triangles and connect them to the vertices.
78  for (iPos += (3 - inc); iPos < iEnd; iPos += inc, ++triangleIdx) {
79  // It should never reallocate or every pointer will be invalid.
80  OgreAssertDbg(data->mTriangleList.capacity() > data->mTriangleList.size(), "");
81  data->mTriangleList.push_back(LodData::Triangle());
82  LodData::Triangle* tri = &data->mTriangleList.back();
83  tri->isRemoved = false;
84  tri->submeshID = submeshID;
85 
86  if(triangleIdx > 0)
87  {
88  switch(renderOp)
89  {
90  case RenderOperation::OT_TRIANGLE_LIST:
91  i0 = iPos[0];
92  i1 = iPos[1];
93  i2 = iPos[2];
94  break;
95  case RenderOperation::OT_TRIANGLE_STRIP:
96  if(triangleIdx & 1)
97  i0 = i2;
98  else
99  i1 = i2;
100  i2 = iPos[0];
101  break;
102  case RenderOperation::OT_TRIANGLE_FAN:
103  i1 = i2;
104  i2 = iPos[0];
105  break;
106  default:
107  OgreAssert(false, "Invalid RenderOperation");
108  break;
109  }
110  }
111 
112  // Invalid index: Index is bigger then vertex buffer size.
113  OgreAssertDbg(i0 < lookup.size() && i1 < lookup.size() && i2 < lookup.size(), "");
114  tri->vertexID[0] = i0;
115  tri->vertexID[1] = i1;
116  tri->vertexID[2] = i2;
117  tri->vertex[0] = lookup[i0];
118  tri->vertex[1] = lookup[i1];
119  tri->vertex[2] = lookup[i2];
120 
121  if (tri->isMalformed()) {
122 #if OGRE_DEBUG_MODE
123  stringstream str;
124  str << "In " << data->mMeshName << " malformed triangle found with ID: " << LodData::getVectorIDFromPointer(data->mTriangleList, tri) << ". " <<
125  std::endl;
126  printTriangle(tri, str);
127  str << "It will be excluded from Lod level calculations.";
128  LogManager::getSingleton().stream() << str.str();
129 #endif
130  tri->isRemoved = true;
131  data->mIndexBufferInfoList[tri->submeshID].indexCount -= 3;
132  continue;
133  }
134  tri->computeNormal();
135  addTriangleToEdges(data, tri);
136  }
137  }
138 };

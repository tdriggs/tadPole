136  struct PassGroupLess
137  {
138  bool operator()(const Pass* a, const Pass* b) const
139  {
140  // Sort by passHash, which is pass, then texture unit changes
141  uint32 hasha = a->getHash();
142  uint32 hashb = b->getHash();
143  if (hasha == hashb)
144  {
145  // Must differentTransparentQueueItemLessiate by pointer incase 2 passes end up with the same hash
146  return a < b;
147  }
148  else
149  {
150  return hasha < hashb;
151  }
152  }

155  struct DepthSortDescendingLess
156  {
157  const Camera* camera;
158 
159  DepthSortDescendingLess(const Camera* cam)
160  : camera(cam)
161  {
162  }
163 
164  bool operator()(const RenderablePass& a, const RenderablePass& b) const
165  {
166  if (a.renderable == b.renderable)
167  {
168  // Same renderable, sort by pass hash
169  return a.pass->getHash() < b.pass->getHash();
170  }
171  else
172  {
173  // Different renderables, sort by depth
174  Real adepth = a.renderable->getSquaredViewDepth(camera);
175  Real bdepth = b.renderable->getSquaredViewDepth(camera);
176  if (Math::RealEqual(adepth, bdepth))
177  {
178  // Must return deterministic result, doesn't matter what
179  return a.pass < b.pass;
180  }
181  else
182  {
183  // Sort DESCENDING by depth (i.e. far objects first)
184  return (adepth > bdepth);
185  }
186  }
187 
188  }

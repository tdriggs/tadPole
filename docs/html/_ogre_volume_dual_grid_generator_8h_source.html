163  mIs->addMarchingCubesTriangles(corners, values, mMb);
164  Vector3 from = mRoot->getFrom();
165  Vector3 to = mRoot->getTo();
166  if (corners[0].z == from.z && corners[0].z != mTotalFrom.z)
167  {
168  mIs->addMarchingSquaresTriangles(corners, values, IsoSurface::MS_CORNERS_BACK, mMaxMSDistance, mMb);
169  }
170  if (corners[2].z == to.z && corners[2].z != mTotalTo.z)
171  {
172  mIs->addMarchingSquaresTriangles(corners, values, IsoSurface::MS_CORNERS_FRONT, mMaxMSDistance, mMb);
173  }
174  if (corners[0].x == from.x && corners[0].x != mTotalFrom.x)
175  {
176  mIs->addMarchingSquaresTriangles(corners, values, IsoSurface::MS_CORNERS_LEFT, mMaxMSDistance, mMb);
177  }
178  if (corners[1].x == to.x && corners[1].x != mTotalTo.x)
179  {
180  mIs->addMarchingSquaresTriangles(corners, values, IsoSurface::MS_CORNERS_RIGHT, mMaxMSDistance, mMb);
181  }
182  if (corners[5].y == to.y && corners[5].y != mTotalTo.y)
183  {
184  mIs->addMarchingSquaresTriangles(corners, values, IsoSurface::MS_CORNERS_TOP, mMaxMSDistance, mMb);
185  }
186  if (corners[0].y == from.y && corners[0].y != mTotalFrom.y)
187  {
188  mIs->addMarchingSquaresTriangles(corners, values, IsoSurface::MS_CORNERS_BOTTOM, mMaxMSDistance, mMb);
189  }

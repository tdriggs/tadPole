61  /*-----------------------------------------------------------------------------
62  | Quits the current sample and starts a new one.
63  -----------------------------------------------------------------------------*/
64  virtual void runSample(Sample* s)
65  {
66 #if OGRE_PROFILING
67  Ogre::Profiler* prof = Ogre::Profiler::getSingletonPtr();
68  if (prof)
69  prof->setEnabled(false);
70 #endif
71 
72  if (mCurrentSample)
73  {
74  mCurrentSample->_shutdown(); // quit current sample
75  mSamplePaused = false; // don't pause the next sample
76  }
77 
78  mWindow->removeAllViewports(); // wipe viewports
79  mWindow->resetStatistics();
80 
81  if (s)
82  {
83  // retrieve sample's required plugins and currently installed plugins
84  Ogre::Root::PluginInstanceList ip = mRoot->getInstalledPlugins();
85  Ogre::StringVector rp = s->getRequiredPlugins();
86 
87  for (Ogre::StringVector::iterator j = rp.begin(); j != rp.end(); j++)
88  {
89  bool found = false;
90  // try to find the required plugin in the current installed plugins
91  for (Ogre::Root::PluginInstanceList::iterator k = ip.begin(); k != ip.end(); k++)
92  {
93  if ((*k)->getName() == *j)
94  {
95  found = true;
96  break;
97  }
98  }
99  if (!found) // throw an exception if a plugin is not found
100  {
101  Ogre::String desc = "Sample requires plugin: " + *j;
102  Ogre::String src = "SampleContext::runSample";
103  OGRE_EXCEPT(Ogre::Exception::ERR_NOT_IMPLEMENTED, desc, src);
104  }
105  }
106 
107  // throw an exception if samples requires the use of another renderer
108  Ogre::String rrs = s->getRequiredRenderSystem();
109  if (!rrs.empty() && rrs != mRoot->getRenderSystem()->getName())
110  {
111  Ogre::String desc = "Sample only runs with renderer: " + rrs;
112  Ogre::String src = "SampleContext::runSample";
113  OGRE_EXCEPT(Ogre::Exception::ERR_INVALID_STATE, desc, src);
114  }
115 
116  // test system capabilities against sample requirements
117  s->testCapabilities(mRoot->getRenderSystem()->getCapabilities());
118 
119  s->_setup(mWindow, mFSLayer, mOverlaySystem); // start new sample
120  }
121 #if OGRE_PROFILING
122  if (prof)
123  prof->setEnabled(true);
124 #endif
125 
126  mCurrentSample = s;
127  }

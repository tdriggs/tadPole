507  typedef std::pair<String,String> MatGroupPair;
508  struct MatGroupPair_less
509  {
510  // ensure we arrange the list first by material name then by group name
511  bool operator()(const MatGroupPair& p1, const MatGroupPair& p2) const
512  {
513  int cmpVal = strcmp(p1.first.c_str(),p2.first.c_str());
514  return (cmpVal < 0) || ((cmpVal == 0) && (strcmp(p1.second.c_str(),p2.second.c_str()) < 0));
515  }
516  };

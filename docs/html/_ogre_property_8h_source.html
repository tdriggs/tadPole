369  void setPropertyImpl(const String& name, const T& val, PropertyType typeCheck)
370  {
371  PropertyBase* baseProp = getProperty(name);
372  if (baseProp->getType() != typeCheck)
373  {
374  StringStream msg;
375  msg << "Property error: type passed in: '" << PropertyDef::getTypeName(typeCheck)
376  << "', type of property: '" << PropertyDef::getTypeName(baseProp->getType()) << "'";
377  OGRE_EXCEPT(Exception::ERR_INVALIDPARAMS, msg.str(), "PropertySet::setPropertyImpl");
378  }

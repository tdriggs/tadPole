72  /*-----------------------------------------------------------------------------
73  | Automatically saves position and orientation for free-look cameras.
74  -----------------------------------------------------------------------------*/
75  virtual void saveState(Ogre::NameValuePairList& state)
76  {
77  if (mCameraMan->getStyle() == CS_FREELOOK)
78  {
79  state["CameraPosition"] = Ogre::StringConverter::toString(mCameraNode->getPosition());
80  state["CameraOrientation"] = Ogre::StringConverter::toString(mCameraNode->getOrientation());
81  }
82  }
83 
84  /*-----------------------------------------------------------------------------
85  | Automatically restores position and orientation for free-look cameras.
86  -----------------------------------------------------------------------------*/
87  virtual void restoreState(Ogre::NameValuePairList& state)
88  {
89  if (state.find("CameraPosition") != state.end() && state.find("CameraOrientation") != state.end())
90  {
91  mCameraMan->setStyle(CS_FREELOOK);
92  mCameraNode->setPosition(Ogre::StringConverter::parseVector3(state["CameraPosition"]));
93  mCameraNode->setOrientation(Ogre::StringConverter::parseQuaternion(state["CameraOrientation"]));
94  }
95  }
96 
97  virtual bool frameRenderingQueued(const Ogre::FrameEvent& evt)
98  {
99  mTrayMgr->frameRendered(evt);
100  mControls->frameRendered(evt);
101 
102  if (!mTrayMgr->isDialogVisible())
103  {
104  mCameraMan->frameRendered(evt); // if dialog isn't up, then update the camera
105  }
106 
107  return true;
108  }

110  virtual bool keyPressed(const KeyboardEvent& evt)
111  {
112  int key = evt.keysym.sym;
113 
114  if (key == 'h' || key == SDLK_F1) // toggle visibility of help dialog
115  {
116  if (!mTrayMgr->isDialogVisible() && mInfo["Help"] != "") mTrayMgr->showOkDialog("Help", mInfo["Help"]);
117  else mTrayMgr->closeDialog();
118  }
119 
120  if (mTrayMgr->isDialogVisible()) return true; // don't process any more keys if dialog is up
121 
122  mControls->keyPressed(evt);
123  mCameraMan->keyPressed(evt);
124  return true;
125  }
126 
127  virtual bool keyReleased(const KeyboardEvent& evt)
128  {
129  mCameraMan->keyReleased(evt);
130 
131  return true;
132  }

179  {

189  virtual bool touchReleased(const TouchFingerEvent& evt) {

192  return mouseReleased(e);

194 
195  virtual bool mouseWheelRolled(const MouseWheelEvent& evt) {

200  /*-----------------------------------------------------------------------------
201  | Extended to setup a default tray interface and camera controller.
202  -----------------------------------------------------------------------------*/
203  virtual void _setup(Ogre::RenderWindow* window, Ogre::FileSystemLayer* fsLayer, Ogre::OverlaySystem* overlaySys)
204  {
205  // assign mRoot here in case Root was initialised after the Sample's constructor ran.
206  mRoot = Ogre::Root::getSingletonPtr();
207  mWindow = window;
208  mFSLayer = fsLayer;
209  mOverlaySystem = overlaySys;
210 
211  locateResources();
212  createSceneManager();
213  setupView();
214 
215  mTrayMgr = new TrayManager("SampleControls", window, this); // create a tray interface
216 
217  loadResources();
218  mResourcesLoaded = true;
219 
220  // show stats and logo and hide the cursor
221  mTrayMgr->showFrameStats(TL_BOTTOMLEFT);
222  mTrayMgr->showLogo(TL_BOTTOMRIGHT);
223  mTrayMgr->hideCursor();
224 
225  mControls = new AdvancedRenderControls(mTrayMgr, mCamera);
226  setupContent();
227  mContentSetup = true;
228 
229  mDone = false;
230  }

232  virtual void _shutdown()
233  {
234  Sample::_shutdown();
235 
236  delete mControls;
237  delete mTrayMgr;
238  delete mCameraMan;
239 
240  // restore settings we may have changed, so as not to affect other samples
241  Ogre::MaterialManager::getSingleton().setDefaultTextureFiltering(Ogre::TFO_BILINEAR);
242  Ogre::MaterialManager::getSingleton().setDefaultAnisotropy(1);
243  }

247  virtual void setupView()
248  {
249  // setup default viewport layout and camera
250  mCamera = mSceneMgr->createCamera("MainCamera");
251  mCameraNode = mSceneMgr->getRootSceneNode()->createChildSceneNode();
252  mCameraNode->attachObject(mCamera);
253  mCameraNode->setFixedYawAxis(true);
254  mViewport = mWindow->addViewport(mCamera);
255  mCamera->setAspectRatio((Ogre::Real)mViewport->getActualWidth() / (Ogre::Real)mViewport->getActualHeight());
256  mCamera->setAutoAspectRatio(true);
257  mCamera->setNearClipDistance(5);
258 
259  mCameraMan = new CameraMan(mCameraNode); // create a default camera controller
260  }
261 
262  virtual void setDragLook(bool enabled)
263  {
264  if (enabled)
265  {
266  mCameraMan->setStyle(CS_MANUAL);

269  }

271  {
272  mCameraMan->setStyle(CS_FREELOOK);
273  mTrayMgr->hideCursor();
274  mDragLook = false;
275  }
276  }
277 
278  Ogre::Viewport* mViewport; // main viewport
279  Ogre::Camera* mCamera; // main camera

282  CameraMan* mCameraMan; // basic camera controller
283  AdvancedRenderControls* mControls; // sample details panel
284  bool mCursorWasVisible; // was cursor visible before dialog appeared
285  bool mDragLook; // click and drag to free-look
286  };

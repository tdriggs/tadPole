232  virtual void* lock(size_t offset, size_t length, LockOptions options, UploadOptions uploadOpt = HBU_DEFAULT)
233  {
234  assert(!isLocked() && "Cannot lock this buffer, it is already locked!");
235 
236  void* ret = NULL;
237  if ((length + offset) > mSizeInBytes)
238  {
239  OGRE_EXCEPT(Exception::ERR_INVALIDPARAMS,
240  "Lock request out of bounds.",
241  "HardwareBuffer::lock");
242  }
243  else if (mUseShadowBuffer)
244  {
245  if (options != HBL_READ_ONLY)
246  {
247  // we have to assume a read / write lock so we use the shadow buffer
248  // and tag for sync on unlock()
249  mShadowUpdated = true;
250  }
251 
252  ret = mShadowBuffer->lock(offset, length, options, uploadOpt);
253  }
254  else
255  {
256  // Lock the real buffer if there is no shadow buffer
257  ret = lockImpl(offset, length, options);
258  mIsLocked = true;
259  }
260  mLockStart = offset;
261  mLockSize = length;
262  mLockUploadOption = uploadOpt;
263  return ret;
264  }

352  virtual void _updateFromShadow(void)
353  {
354  if (mUseShadowBuffer && mShadowUpdated && !mSuppressHardwareUpdate)
355  {
356  // Do this manually to avoid locking problems
357  const void *srcData = mShadowBuffer->lockImpl(
358  mLockStart, mLockSize, HBL_READ_ONLY);
359  // Lock with discard if the whole buffer was locked, otherwise normal
360  LockOptions lockOpt;
361  if (mLockStart == 0 && mLockSize == mSizeInBytes)
362  lockOpt = HBL_DISCARD;
363  else
364  lockOpt = HBL_NORMAL;
365 
366  void *destData = this->lockImpl(
367  mLockStart, mLockSize, lockOpt);
368  // Copy shadow to real
369  memcpy(destData, srcData, mLockSize);
370  this->unlockImpl();
371  mShadowBuffer->unlockImpl();
372  mShadowUpdated = false;
373  }
374  }

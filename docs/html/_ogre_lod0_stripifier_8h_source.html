60  struct Lod0Stripifier::RemapInfo
61  {
62  RemapInfo() : usedCount(0) { }
63  void prepare(unsigned originalSize) { usedCount = 0; indexMap.resize(originalSize, UnusedIdx); }
64  template<typename IDX> void markUsedIndices(IDX* indices, size_t indexCount)
65  {
66  for(IDX *idx = indices, *idx_end = idx + indexCount; idx < idx_end; ++idx)
67  if(indexMap[*idx] == UnusedIdx)
68  indexMap[*idx] = usedCount++;
69  }
70  void renumerate()
71  {
72  usedCount = 0;
73  for(size_t idx = 0, idx_end = indexMap.size(); idx < idx_end; ++idx)
74  if(indexMap[idx] != UnusedIdx)
75  indexMap[idx] = usedCount++;
76  }
77  bool nothingToStrip() const { return usedCount == indexMap.size(); }
78  HardwareIndexBuffer::IndexType minimalIndexType() const { return usedCount < 0xFFFF ? HardwareIndexBuffer::IT_16BIT : HardwareIndexBuffer::IT_32BIT; }
79 
80  public:
81  enum{ UnusedIdx = (unsigned)-1 };
82  vector<unsigned>::type indexMap; // returns new index if indexed by old index, or UnusedIdx
83  unsigned usedCount;
84  };

93  inline void Lod0Stripifier::generateRemapInfo(const MeshPtr& mesh, bool stableVertexOrder)
94  {
95  size_t submeshCount = mesh->getNumSubMeshes();
96  remapInfos.resize(1 + submeshCount);
97  remapInfos[0].prepare(mesh->sharedVertexData ? mesh->sharedVertexData->vertexCount : 0);
98  for(ushort i = 0; i < submeshCount; i++)
99  {
100  const SubMesh* submesh = mesh->getSubMesh(i);
101  remapInfos[1 + i].prepare(submesh->useSharedVertices ? 0 : submesh->vertexData->vertexCount);
102 
103  RemapInfo& remapInfo = remapInfos[submesh->useSharedVertices ? 0 : 1 + i];
104 
105  for(ushort lod = mesh->getNumLodLevels() - 1; lod != 0; --lod) // intentionally skip lod0, visit in reverse order to improve vertex locality for high lods
106  {
107  IndexData *lodIndexData = submesh->mLodFaceList[lod - 1];
108  void* ptr = lodIndexData->indexBuffer->lock(
109  lodIndexData->indexStart * lodIndexData->indexBuffer->getIndexSize(),
110  lodIndexData->indexCount * lodIndexData->indexBuffer->getIndexSize(),
111  HardwareBuffer::HBL_READ_ONLY);
112 
113  if(lodIndexData->indexBuffer->getType() == HardwareIndexBuffer::IT_32BIT)
114  remapInfo.markUsedIndices((uint32*)ptr, lodIndexData->indexCount);
115  else
116  remapInfo.markUsedIndices((uint16*)ptr, lodIndexData->indexCount);
117 
118  lodIndexData->indexBuffer->unlock();
119  }
120 
121  if(stableVertexOrder)
122  remapInfos[1 + i].renumerate();
123  }
124  if(stableVertexOrder)
125  remapInfos[0].renumerate();
126  }

128  inline void Lod0Stripifier::performIndexDataRemap(IndexData* indexData, const RemapInfo& remapInfo)
129  {
130  if(remapInfo.nothingToStrip())
131  return;
132 
133  size_t indexCount = indexData->indexCount;
134  HardwareIndexBuffer::IndexType indexType = indexData->indexBuffer->getType();
135  HardwareIndexBuffer::IndexType newIndexType = remapInfo.minimalIndexType();
136  HardwareIndexBufferSharedPtr newIndexBuffer =
137  HardwareBufferManager::getSingleton().createIndexBuffer(
138  newIndexType, indexCount, indexData->indexBuffer->getUsage(), indexData->indexBuffer->hasShadowBuffer());
139 
140  void* pSrc = indexData->indexBuffer->lock(
141  indexData->indexStart * indexData->indexBuffer->getIndexSize(),
142  indexData->indexCount * indexData->indexBuffer->getIndexSize(),
143  HardwareBuffer::HBL_READ_ONLY);
144  void* pDst = newIndexBuffer->lock(HardwareBuffer::HBL_DISCARD);
145 
146  if(indexType == HardwareIndexBuffer::IT_32BIT && newIndexType == HardwareIndexBuffer::IT_32BIT)
147  {
148  uint32 *pSrc32 = (uint32*)pSrc, *pDst32 = (uint32*)pDst;
149  for(size_t i = 0; i < indexCount; ++i)
150  pDst32[i] = remapInfo.indexMap[pSrc32[i]];
151  }
152  if(indexType == HardwareIndexBuffer::IT_32BIT && newIndexType == HardwareIndexBuffer::IT_16BIT)
153  {
154  uint32 *pSrc32 = (uint32*)pSrc; uint16 *pDst16 = (uint16*)pDst;
155  for(size_t i = 0; i < indexCount; ++i)
156  pDst16[i] = (uint16)remapInfo.indexMap[pSrc32[i]];
157  }
158  if(indexType == HardwareIndexBuffer::IT_16BIT && newIndexType == HardwareIndexBuffer::IT_32BIT)
159  {
160  uint16 *pSrc16 = (uint16*)pSrc; uint32 *pDst32 = (uint32*)pDst;
161  for(size_t i = 0; i < indexCount; ++i)
162  pDst32[i] = remapInfo.indexMap[pSrc16[i]];
163  }
164  else if(indexType == HardwareIndexBuffer::IT_16BIT && newIndexType == HardwareIndexBuffer::IT_16BIT)
165  {
166  uint16 *pSrc16 = (uint16*)pSrc, *pDst16 = (uint16*)pDst;
167  for(size_t i = 0; i < indexCount; ++i)
168  pDst16[i] = uint16(remapInfo.indexMap[pSrc16[i]]);
169  }
170 
171  indexData->indexBuffer->unlock();
172  newIndexBuffer->unlock();
173 
174  indexData->indexBuffer = newIndexBuffer;
175  indexData->indexStart = 0;
176  }

178  inline void Lod0Stripifier::performVertexDataRemap(VertexData* vertexData, const RemapInfo& remapInfo)
179  {
180  if(remapInfo.nothingToStrip())
181  return;
182 
183  // Copy vertex buffers in turn
184  typedef std::map<HardwareVertexBufferSharedPtr, HardwareVertexBufferSharedPtr> VBMap;
185  VBMap alreadyProcessed; // prevent duplication of the same buffer bound under several indices
186  const VertexBufferBinding::VertexBufferBindingMap& bindings = vertexData->vertexBufferBinding->getBindings();
187  VertexBufferBinding::VertexBufferBindingMap::const_iterator vbi, vbend;
188  for(vbi = bindings.begin(), vbend = bindings.end(); vbi != vbend; ++vbi)
189  {
190  HardwareVertexBufferSharedPtr srcbuf = vbi->second;
191  VBMap::iterator it = alreadyProcessed.find(srcbuf);
192  if(it != alreadyProcessed.end())
193  {
194  vertexData->vertexBufferBinding->setBinding(vbi->first, it->second);
195  continue;
196  }
197 
198  HardwareVertexBufferSharedPtr dstbuf = getRemappedVertexBuffer(srcbuf, vertexData->vertexStart, vertexData->vertexCount, remapInfo);
199  vertexData->vertexBufferBinding->setBinding(vbi->first, dstbuf);
200  alreadyProcessed[srcbuf] = dstbuf;
201  }
202 
203  vertexData->vertexStart = 0;
204  vertexData->vertexCount = remapInfo.usedCount;
205 
206  vertexData->hardwareShadowVolWBuffer = HardwareVertexBufferSharedPtr(); // TODO: check this
207  vertexData->hwAnimationDataList.clear(); // TODO: check this
208  vertexData->hwAnimDataItemsUsed = 0; // TODO: check this
209  }

211  inline HardwareVertexBufferSharedPtr Lod0Stripifier::getRemappedVertexBuffer(
212  HardwareVertexBufferSharedPtr srcbuf, size_t srcStart, size_t srcCount, const RemapInfo& remapInfo)
213  {
214  assert(!remapInfo.nothingToStrip());
215 
216  size_t vertexSize = srcbuf->getVertexSize();
217  HardwareVertexBufferSharedPtr dstbuf =
218  HardwareBufferManager::getSingleton().createVertexBuffer(
219  vertexSize, remapInfo.usedCount, srcbuf->getUsage(), srcbuf->hasShadowBuffer());
220 
221  char* pSrc = (char*)srcbuf->lock(srcStart * vertexSize, srcCount * vertexSize, HardwareBuffer::HBL_READ_ONLY);
222  char* pDst = (char*)dstbuf->lock(HardwareBuffer::HBL_DISCARD);
223 
224  for(size_t oldIdx = 0, oldIdxEnd = remapInfo.indexMap.size(); oldIdx < oldIdxEnd; ++oldIdx)
225  {
226  unsigned newIdx = remapInfo.indexMap[oldIdx];
227  if(newIdx != RemapInfo::UnusedIdx)
228  memcpy(pDst + newIdx * vertexSize, pSrc + oldIdx * vertexSize, vertexSize);
229  }
230 
231  srcbuf->unlock();
232  dstbuf->unlock();
233  return dstbuf;
234  }

236  template<class MeshOrSubmesh>
237  inline void Lod0Stripifier::performBoneAssignmentRemap(MeshOrSubmesh* m, const RemapInfo& remapInfo)
238  {
239  if(remapInfo.nothingToStrip() || m->getBoneAssignments().empty())
240  return;
241 
242  Mesh::VertexBoneAssignmentList tmp = m->getBoneAssignments();
243  m->clearBoneAssignments();
244  for(Mesh::VertexBoneAssignmentList::const_iterator it = tmp.begin(), it_end = tmp.end(); it != it_end; ++it)
245  {
246  VertexBoneAssignment vba = it->second;
247  unsigned newIdx = remapInfo.indexMap[vba.vertexIndex];
248  if(newIdx != RemapInfo::UnusedIdx)
249  {
250  vba.vertexIndex = newIdx;
251  m->addBoneAssignment(vba);
252  }
253  }
254  }

256  inline void Lod0Stripifier::performPoseRemap(Pose* pose, const RemapInfo& remapInfo)
257  {
258  if(remapInfo.nothingToStrip() || (pose->getVertexOffsets().empty() && pose->getNormals().empty()))
259  return;
260 
261  Pose::VertexOffsetMap vv = pose->getVertexOffsets();
262  Pose::NormalsMap nn = pose->getNormals();
263  pose->clearVertices();
264  for(Pose::VertexOffsetMap::const_iterator vit = vv.begin(), vit_end = vv.end(); vit != vit_end; ++vit)
265  {
266  unsigned newIdx = remapInfo.indexMap[vit->first];
267  if(newIdx != RemapInfo::UnusedIdx)
268  {
269  if(pose->getIncludesNormals())
270  pose->addVertex(newIdx, vit->second, nn[vit->first]);
271  else
272  pose->addVertex(newIdx, vit->second);
273  }
274  }
275  }

293  inline bool Lod0Stripifier::StripLod0Vertices(const MeshPtr& mesh, bool stableVertexOrder)
294  {
295  // we need at least one lod level except lod0 that would be stripped
296  ushort numLods = mesh->hasManualLodLevel() ? 1 : mesh->getNumLodLevels();
297  if(numLods <= 1)
298  return false;
299 
300  bool edgeListWasBuilt = mesh->isEdgeListBuilt();
301  mesh->freeEdgeList();
302 
303  generateRemapInfo(mesh, stableVertexOrder);
304 
305  if(mesh->sharedVertexData)
306  performVertexDataRemap(mesh->sharedVertexData, remapInfos[0]);
307  performBoneAssignmentRemap(mesh.get(), remapInfos[0]);
308 
309  size_t submeshCount = mesh->getNumSubMeshes();
310  for(size_t i = 0; i < submeshCount; i++)
311  {
312  SubMesh* submesh = mesh->getSubMesh(i);
313  const RemapInfo& remapInfo = remapInfos[submesh->useSharedVertices ? 0 : 1 + i];
314 
315  if(!submesh->useSharedVertices)
316  performVertexDataRemap(submesh->vertexData, remapInfo);
317  performBoneAssignmentRemap(submesh, remapInfo);
318 
319  for(ushort lod = numLods - 1; lod != 0; --lod) // intentionally skip lod0
320  {
321  IndexData *lodIndexData = submesh->mLodFaceList[lod - 1]; // lod0 is stored separately
322  performIndexDataRemap(lodIndexData, remapInfo);
323  }
324 
325  OGRE_DELETE submesh->indexData;
326  submesh->indexData = submesh->mLodFaceList[0];
327  submesh->mLodFaceList.erase(submesh->mLodFaceList.begin());
328  }
329 
330  for(ushort lod = 1; lod < numLods - 1; ++lod)
331  mesh->_setLodUsage(lod, mesh->getLodLevel(lod + 1));
332  mesh->_setLodInfo(numLods - 1);
333 
334 
335  PoseList::const_iterator it;
336  for( it = mesh->getPoseList().begin(); it != mesh->getPoseList().end(); ++it)
337  {
338  Pose* pose = *it;
339  performPoseRemap(pose, remapInfos[pose->getTarget()]);
340  }
341 
342  for(unsigned short a = 0; a < mesh->getNumAnimations(); ++a)
343  {
344  Animation* anim = mesh->getAnimation(a);
345  Animation::VertexTrackIterator trackIt = anim->getVertexTrackIterator();
346  while(trackIt.hasMoreElements())
347  {
348  VertexAnimationTrack* track = trackIt.getNext();
349  performAnimationTrackRemap(track, remapInfos[track->getHandle()]);
350  }
351  }
352 
353  if(edgeListWasBuilt)
354  mesh->buildEdgeList();
355 
356  return true;
357  }
358 }
